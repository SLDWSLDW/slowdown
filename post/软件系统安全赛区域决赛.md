## rsa

### Level 1

20 个公钥 + 10 个密文 + Asmuth-Bloom 秘密共享

| 攻击  | 密钥  | 原理  |
| --- | --- | --- |
| GCD 共因子 | key-1&2, key-4&15 | `gcd(n1,n2)=p` 直接分解 |
| Wiener 攻击 | key-6,12,17 | e 极大 → d=65537 极小，连分数恢复 |
| Fermat 分解 | key-3,7,13,18 | p≈q，从 √n 附近找因子 |
| Pollard p-1 | key-5 | p-1 最大素因子=457，B=10000 即可 |

解密 7 个密文 → 收集 7 个 shares → CRT 重建 message7

**密码：**`9Zr4M1ThwVCHe4nHnmOcilJ8`

```python
from Crypto.PublicKey import RSA
from Crypto.Cipher import AES, PKCS1_OAEP
from Crypto.Util.number import bytes_to_long, long_to_bytes
from math import gcd, isqrt
import os

# ===== 工具函数 =====
def modinv(a, m):
    g, x, _ = extended_gcd(a, m)
    if g != 1:
        raise ValueError("No inverse")
    return x % m

def extended_gcd(a, b):
    if a == 0:
        return b, 0, 1
    g, x, y = extended_gcd(b % a, a)
    return g, y - (b // a) * x, x

def reconstruct_private_key(n, e, p, q):
    phi = (p - 1) * (q - 1)
    d = modinv(e, phi)
    return RSA.construct((n, e, d, p, q))

# ===== 加载所有公钥 =====
keys = []
for i in range(20):
    with open(f'key-{i}.pem', 'r') as f:
        keys.append(RSA.import_key(f.read()))

# ===== GCD 攻击 =====
private_keys = {}

# key-1 和 key-2 共享因子
p12 = gcd(keys[1].n, keys[2].n)
q1 = keys[1].n // p12
q2 = keys[2].n // p12
private_keys[1] = reconstruct_private_key(keys[1].n, keys[1].e, p12, q1)
private_keys[2] = reconstruct_private_key(keys[2].n, keys[2].e, p12, q2)
print(f"[+] key-1 factored via GCD")
print(f"[+] key-2 factored via GCD")

# key-4 和 key-15 共享因子
p415 = gcd(keys[4].n, keys[15].n)
q4 = keys[4].n // p415
q15 = keys[15].n // p415
private_keys[4] = reconstruct_private_key(keys[4].n, keys[4].e, p415, q4)
private_keys[15] = reconstruct_private_key(keys[15].n, keys[15].e, p415, q15)
print(f"[+] key-4 factored via GCD")
print(f"[+] key-15 factored via GCD")

# ===== Wiener 攻击 =====
def continued_fraction(n, d):
    """计算 n/d 的连分数展开"""
    while d:
        yield n // d
        n, d = d, n % d

def convergents(cf):
    """从连分数系数生成收敛分数"""
    n0, n1 = 1, 0
    d0, d1 = 0, 1
    for a in cf:
        n0, n1 = a * n0 + n1, n0
        d0, d1 = a * d0 + d1, d0
        yield n0, d0

def wiener_attack(e, n):
    """Wiener 攻击：当 d 较小时恢复私钥"""
    for k, d in convergents(continued_fraction(e, n)):
        if k == 0:
            continue
        # 检验 d 是否正确：e*d ≡ 1 (mod φ(n))
        # φ(n) = (e*d - 1) / k
        if (e * d - 1) % k != 0:
            continue
        phi = (e * d - 1) // k
        # 解方程 x^2 - (n - phi + 1)x + n = 0
        b = n - phi + 1
        discriminant = b * b - 4 * n
        if discriminant < 0:
            continue
        sqrt_disc = isqrt(discriminant)
        if sqrt_disc * sqrt_disc != discriminant:
            continue
        p = (b + sqrt_disc) // 2
        q = (b - sqrt_disc) // 2
        if p * q == n and p > 1 and q > 1:
            return d, p, q
    return None

# ===== Fermat 分解 =====
def fermat_factor(n, max_iter=500000):
    a = isqrt(n)
    if a * a < n:
        a += 1
    for _ in range(max_iter):
        b2 = a * a - n
        b = isqrt(b2)
        if b * b == b2:
            return a - b, a + b
        a += 1
    return None

for idx in [3, 7, 13, 18]:
    result = fermat_factor(keys[idx].n)
    if result:
        p, q = result
        private_keys[idx] = reconstruct_private_key(keys[idx].n, keys[idx].e, p, q)
        print(f"[+] key-{idx} factored via Fermat (p-q={abs(p-q)})")

# ===== Pollard p-1 =====
def pollard_pm1(n, B=100000):
    a = 2
    for j in range(2, B + 1):
        a = pow(a, j, n)
        if j % 10000 == 0:
            g = gcd(a - 1, n)
            if 1 < g < n:
                return g, n // g
    g = gcd(a - 1, n)
    if 1 < g < n:
        return g, n // g
    return None

# key-5: e=3, gcd(3,q-1)=3, 需要特殊解密，先保存因子
key5_factors = None
for idx in [5]:
    result = pollard_pm1(keys[idx].n)
    if result:
        p, q = result
        key5_factors = (p, q)
        print(f"[+] key-{idx} factored via Pollard p-1 (e=3, special handling needed)")

# ===== Wiener 攻击 =====
for idx in [6, 12, 17]:
    result = wiener_attack(keys[idx].e, keys[idx].n)
    if result:
        d, p, q = result
        private_keys[idx] = reconstruct_private_key(keys[idx].n, keys[idx].e, p, q)
        print(f"[+] key-{idx} cracked via Wiener attack, d={d}")
    else:
        print(f"[-] key-{idx} Wiener attack failed")

print(f"\n[*] Cracked keys: {sorted(private_keys.keys())}")

# ===== 解密函数 =====
def try_decrypt(private_key, ciphertext_data):
    n = private_key.n
    n_bits = n.bit_length()
    key_len = (n_bits + 7) // 8

    if len(ciphertext_data) < key_len + 12 + 16:
        return None

    msg_header = ciphertext_data[:key_len]
    nonce = ciphertext_data[key_len:key_len + 12]
    msg_body = ciphertext_data[key_len + 12:-16]
    tag = ciphertext_data[-16:]

    try:
        if n_bits >= 2048:
            symmetric_key = PKCS1_OAEP.new(private_key).decrypt(msg_header)
        else:
            encrypted_key_int = bytes_to_long(msg_header)
            key_int = pow(encrypted_key_int, private_key.d, n)
            symmetric_key = long_to_bytes(key_int, 16)

        cipher = AES.new(symmetric_key, mode=AES.MODE_GCM, nonce=nonce)
        plaintext = cipher.decrypt_and_verify(msg_body, tag)
        return plaintext
    except Exception:
        return None

# ===== 尝试解密所有密文 =====
print("\n[*] Trying to decrypt ciphertexts...")
decrypted = {}

for ct_idx in range(1, 11):
    with open(f'ciphertext-{ct_idx}.bin', 'rb') as f:
        ct_data = f.read()

    for key_idx, priv_key in private_keys.items():
        result = try_decrypt(priv_key, ct_data)
        if result is not None:
            print(f"[+] ciphertext-{ct_idx} decrypted with key-{key_idx}")
            decrypted[ct_idx] = (key_idx, result)
            break
    else:
        print(f"[-] ciphertext-{ct_idx}: no matching key found")

print(f"\n[*] Decrypted {len(decrypted)}/10 ciphertexts")

# ===== key-5 特殊解密 (e=3, gcd(3,q-1)=3) =====
if key5_factors:
    p5, q5 = key5_factors
    n5 = keys[5].n
    key_len5 = (n5.bit_length() + 7) // 8

    def cube_roots_mod_prime(c, p):
        """求 x^3 = c mod p 的所有解"""
        if c % p == 0:
            return [0]
        if (p - 1) % 3 != 0:
            # gcd(3, p-1) = 1, 唯一解
            d = pow(3, -1, p - 1)
            return [pow(c, d, p)]
        # p-1 = 3^s * t
        s, t = 0, p - 1
        while t % 3 == 0:
            s += 1
            t //= 3
        # 找一个非立方剩余 g
        for g in range(2, p):
            if pow(g, (p - 1) // 3, p) != 1:
                break
        # Adleman-Manders-Miller
        b = pow(g, t, p)
        r = pow(c, (t + 1) // 3 if (t % 3 == 2) else t // 3, p)
        # 对于 q, 用 Tonelli-Shanks 变体
        inv3 = pow(3, -1, p - 1) if gcd(3, p - 1) == 1 else None
        if inv3:
            return [pow(c, inv3, p)]
        omega = pow(g, (p - 1) // 3, p)
        roots = []
        if pow(c, (p - 1) // 3, p) != 1:
            return []
        e_inv = pow(3, -1, t) if gcd(3, t) == 1 else None
        if e_inv:
            x0 = pow(c, e_inv, p)
            # 修正高阶部分
            for _ in range(s):
                while pow(x0, 3, p) != c % p:
                    x0 = x0 * b % p
            roots = [x0 % p, x0 * omega % p, x0 * omega * omega % p]
            return [r % p for r in roots]
        return []

    for ct_idx in range(1, 11):
        if ct_idx in decrypted:
            continue
        with open(f'ciphertext-{ct_idx}.bin', 'rb') as f:
            ct_data = f.read()
        if len(ct_data) < key_len5 + 12 + 16:
            continue
        msg_header = ct_data[:key_len5]
        nonce = ct_data[key_len5:key_len5 + 12]
        msg_body = ct_data[key_len5 + 12:-16]
        tag = ct_data[-16:]
        encrypted_key_int = bytes_to_long(msg_header)
        if encrypted_key_int >= n5:
            continue
        # 求 m^3 = encrypted_key_int mod n5
        # mod p5: gcd(3,p5-1)=1
        roots_p = cube_roots_mod_prime(encrypted_key_int, p5)
        # mod q5: gcd(3,q5-1)=3
        roots_q = cube_roots_mod_prime(encrypted_key_int, q5)
        if not roots_p or not roots_q:
            continue
        # CRT 组合
        for rp in roots_p:
            for rq in roots_q:
                m = (rp * (q5 * pow(q5, -1, p5)) + rq * (p5 * pow(p5, -1, q5))) % n5
                sym_key = long_to_bytes(m, 16)
                try:
                    cipher = AES.new(sym_key, mode=AES.MODE_GCM, nonce=nonce)
                    plaintext = cipher.decrypt_and_verify(msg_body, tag)
                    print(f"[+] ciphertext-{ct_idx} decrypted with key-5 (cube root)")
                    decrypted[ct_idx] = (5, plaintext)
                    break
                except Exception:
                    continue
            else:
                continue
            break
    print(f"[*] After key-5: Decrypted {len(decrypted)}/10 ciphertexts")

# ===== 解析明文，提取 Asmuth-Bloom shares =====
# 每个明文格式：
#   第1行: message1 内容
#   第2-10行: 对应 message2-10 的 share，格式 "di:ki:original_bits"（十六进制）
# 第 j 行（0-indexed）对应 message(j+1)，share 索引 = ciphertext 编号（1-indexed）

# shares_by_msg[msg_idx][ct_idx] = (di, ki, original_bits)
shares_by_msg = {i: {} for i in range(2, 11)}
message1 = None

for ct_idx, (key_idx, plaintext) in sorted(decrypted.items()):
    lines = plaintext.decode('utf-8').strip().split('\n')
    if message1 is None:
        message1 = lines[0]
    # lines[1..9] 对应 message2..10
    for line_i, line in enumerate(lines[1:], start=2):
        line = line.strip()
        if not line:
            continue
        parts = line.split(':')
        if len(parts) == 3:
            di = int(parts[0], 16)
            ki = int(parts[1], 16)
            orig_bits = int(parts[2], 16)
            shares_by_msg[line_i][ct_idx] = (di, ki, orig_bits)

print(f"\n[*] message1: {message1}")
print(f"[*] Shares collected per message:")
for msg_i in range(2, 11):
    print(f"    message{msg_i}: {len(shares_by_msg[msg_i])} shares (need {msg_i})")

# ===== CRT 重建 =====
def crt_reconstruct(shares):
    """shares: list of (di, ki)"""
    M = 1
    for di, ki in shares:
        M *= di
    result = 0
    for di, ki in shares:
        Mi = M // di
        yi = pow(Mi, -1, di)
        result += ki * Mi * yi
    return result % M

print("\n[*] Reconstructing messages via CRT...")
for msg_i in range(2, 11):
    available = shares_by_msg[msg_i]
    if len(available) >= msg_i:
        # 取前 msg_i 个 share
        selected = list(available.values())[:msg_i]
        orig_bits = selected[0][2]
        shares_pairs = [(di, ki) for di, ki, _ in selected]
        S = crt_reconstruct(shares_pairs)
        byte_len = (orig_bits + 7) // 8
        try:
            msg_bytes = S.to_bytes(byte_len, 'big')
            print(f"\n[+] message{msg_i}: {msg_bytes.decode('utf-8')}")
        except Exception as ex:
            print(f"\n[-] message{msg_i}: decode error: {ex}, raw={S}")
    else:
        print(f"[-] message{msg_i}: only {len(available)} shares, need {msg_i}")
```

### Level 2

小私钥 d=180 位 + 使用 λ(n) 而非 φ(n)

```python
e·d ≡ 1 mod λ(n)，λ(n) = lcm(p-1, q-1) = φ(n)/gcd(p-1,q-1)
```

当 `gcd(p-1,q-1)=4` 时，`e·(4d) = 4 + k·φ(n)`，对 `e/n` 做连分数展开，枚举 g∈{1,2,3,4,6} 即可恢复 φ(n) 并分解 n。

### Level 3

复杂泄漏函数 + 双重 Hensel Lifting

```python
leak = (p*C1 ^ q*C2 ^ (p&q)<<64 ^ (p|q)<<48 ^ (p^q)*C3) + (p+q)%2^128 ^ (n & MASK64)
```

**关键性质**：`A mod 2^k` 只依赖 `p mod 2^k`（因为模运算的局部性）

**Step 1**：Hensel Lifting 找 `p mod 2^128`

- 目标方程：`(A_low128 + (p+q) mod 2^128) mod 2^128 ^ (n mod 2^64) = leak mod 2^128`
- 从 `p mod 2 = 1` 逐位提升，每步两个候选，方程唯一确定正确值

**Step 2**：恢复完整 A

- `X = (leak & ~MASK64) | ((leak ^ N_low64) & MASK64)`
- `A = X - (p_low128 + q_low128) mod 2^128`

**Step 3**：Hensel Lifting 找完整 p（1536 位）

- 目标方程：`A mod 2^k = (p*C1 ^ q*C2 ^ ...) mod 2^k`
- 从低位逐位提升，唯一确定完整 p

**Flag：**`dart{379c9308-e9a8-45a1-bd55-45bbd822e86d}`

```python
    return t1 ^ t2 ^ t3 ^ t4 ^ t5
```

## steganography

1.  文件分析 — `steganography_challenge` 是带 35 字节自定义头的 PNG 文件
2.  LSB 隐写 — 从 PNG 像素 RGB 通道的最低有效位提取出一个 ZIP 文件（1551 字节，大小存在头部 `0f060000`）
3.  ZIP 结构 — 外层 ZIP 包含 `flag.zip` 和 `pass1-6.zip`，均加密
4.  CRC 爆破 — `pass*.zip` 内各有 4 字节文件，通过 CRC 碰撞枚举所有可打印字符组合，还原明文：
    - `data3.txt = c1!x`
    - `data4.txt = xtLf`
    - `data5.txt = %fXY`
    - `data6.txt = PkaA`
5.  flag.zip 解密 — 密码为 `c1!xxtLf%fXYPkaA`
6.  零宽字符隐写 — `flag.txt` 内容 "flag is here" 后跟大量 U+200B（零宽空格=0）和 U+200C（零宽非连接符=1），解码二进制得到最终 flag

```python
import struct
import zlib
import zipfile
import io
import string

CHALLENGE_FILE = "steganography_challenge"

# ─────────────────────────────────────────────
# Step 1: 提取 PNG 数据
# ─────────────────────────────────────────────
def extract_png(filepath):
    with open(filepath, "rb") as f:
        data = f.read()
    png_sig = b"\x89PNG\r\n\x1a\n"
    idx = data.find(png_sig)
    assert idx >= 0, "PNG signature not found"
    print(f"[+] PNG signature at offset {idx} (header = {data[:idx].hex()})")
    return data[idx:]


# ─────────────────────────────────────────────
# Step 2: 解析 IDAT 块，解压像素数据
# ─────────────────────────────────────────────
def collect_idat(png_data):
    idat_raw = b""
    pos = 0
    while True:
        p = png_data.find(b"IDAT", pos)
        if p == -1:
            break
        length = struct.unpack(">I", png_data[p - 4 : p])[0]
        chunk_data = png_data[p + 4 : p + 4 + length]
        idat_raw += chunk_data
        print(f"[+] IDAT at type_offset={p}, length={length}")
        pos = p + 4 + length + 4  # skip data + CRC
    return idat_raw


def decompress_idat(idat_raw):
    d = zlib.decompressobj()
    raw = d.decompress(idat_raw)
    print(f"[+] Decompressed {len(raw)} bytes (expected ~{1024*(1+1024*3)})")
    return raw


# ─────────────────────────────────────────────
# Step 3: 应用 PNG 滤波器，还原像素
# ─────────────────────────────────────────────
def paeth(a, b, c):
    p = a + b - c
    pa, pb, pc = abs(p - a), abs(p - b), abs(p - c)
    if pa <= pb and pa <= pc:
        return a
    elif pb <= pc:
        return b
    return c


def reconstruct_pixels(raw, width=1024, height=1024, channels=3):
    pixels = bytearray(width * height * channels)
    prev = bytearray(width * channels)
    pos = 0
    for y in range(height):
        ft = raw[pos] & 0x07          # mask 无效 filter type
        row = bytearray(raw[pos + 1 : pos + 1 + width * channels])
        pos += 1 + width * channels
        if ft == 1:
            for x in range(channels, len(row)):
                row[x] = (row[x] + row[x - channels]) & 0xFF
        elif ft == 2:
            for x in range(len(row)):
                row[x] = (row[x] + prev[x]) & 0xFF
        elif ft == 3:
            for x in range(len(row)):
                a = row[x - channels] if x >= channels else 0
                row[x] = (row[x] + (a + prev[x]) // 2) & 0xFF
        elif ft == 4:
            for x in range(len(row)):
                a = row[x - channels] if x >= channels else 0
                b = prev[x]
                c = prev[x - channels] if x >= channels else 0
                row[x] = (row[x] + paeth(a, b, c)) & 0xFF
        pixels[y * width * channels : (y + 1) * width * channels] = row
        prev = row
    print(f"[+] Pixel reconstruction done, {len(pixels)} bytes")
    return pixels


# ─────────────────────────────────────────────
# Step 4: LSB 提取 ZIP
# ─────────────────────────────────────────────
def extract_lsb_zip(pixels):
    bits = [pixels[i] & 1 for i in range(len(pixels))]
    lsb_bytes = bytearray()
    for i in range(0, len(bits) - 7, 8):
        byte = 0
        for j in range(8):
            byte = (byte << 1) | bits[i + j]
        lsb_bytes.append(byte)

    # 前4字节为小端 uint32 = ZIP 数据长度
    zip_len = struct.unpack("<I", lsb_bytes[:4])[0]
    print(f"[+] LSB hidden data length: {zip_len} bytes")
    assert lsb_bytes[4:6] == b"PK", "ZIP signature not found in LSB data"
    zip_data = bytes(lsb_bytes[4 : 4 + zip_len])
    print(f"[+] Extracted ZIP ({len(zip_data)} bytes)")
    return zip_data


# ─────────────────────────────────────────────
# Step 5: CRC-32 碰撞还原 data*.txt 明文
# ─────────────────────────────────────────────
def crack_crc(target_crcs: dict) -> dict:
    found = {}
    printable = string.printable[:95]   # 所有可打印 ASCII
    total = len(target_crcs)
    print(f"[*] Brute-forcing CRC-32 for {total} files (4-char printable)...")
    for a in printable:
        for b in printable:
            for c in printable:
                for d in printable:
                    content = (a + b + c + d).encode()
                    crc = zlib.crc32(content) & 0xFFFFFFFF
                    for idx, target in target_crcs.items():
                        if idx not in found and crc == target:
                            found[idx] = a + b + c + d
                            print(f"[+] data{idx}.txt = {repr(a+b+c+d)}")
                if len(found) == total:
                    break
            if len(found) == total:
                break
        if len(found) == total:
            break
    return found


# ─────────────────────────────────────────────
# Step 6: 解密 flag.zip，解码零宽字符
# ─────────────────────────────────────────────
def decode_zwc(text: str) -> str:
    bits = []
    for ch in text:
        if ch == "\u200b":
            bits.append("0")
        elif ch == "\u200c":
            bits.append("1")
    bit_str = "".join(bits)
    result = ""
    for i in range(0, len(bit_str) - 7, 8):
        result += chr(int(bit_str[i : i + 8], 2))
    return result


# ─────────────────────────────────────────────
# Main
# ─────────────────────────────────────────────
def main():
    print("=" * 60)
    print("  Steganography Challenge Exploit")
    print("=" * 60)

    # Step 1
    png_data = extract_png(CHALLENGE_FILE)

    # Step 2
    idat_raw = collect_idat(png_data)
    raw = decompress_idat(idat_raw)

    # Step 3
    pixels = reconstruct_pixels(raw)

    # Step 4
    zip_data = extract_lsb_zip(pixels)
    outer_zip = zipfile.ZipFile(io.BytesIO(zip_data))
    print(f"[+] Outer ZIP contents: {outer_zip.namelist()}")

    # Step 5: 读取 pass*.zip 的 CRC
    target_crcs = {}
    for i in range(1, 7):
        inner_data = outer_zip.read(f"pass{i}.zip")
        zf = zipfile.ZipFile(io.BytesIO(inner_data))
        info = zf.getinfo(f"data{i}.txt")
        target_crcs[i] = info.CRC
        print(f"[*] pass{i}.zip -> data{i}.txt CRC = 0x{info.CRC:08x}")

    plaintext_map = crack_crc(target_crcs)

    # 拼接密码：data3+data4+data5+data6（data1/2 为干扰项）
    import itertools
    flag_zip_data = outer_zip.read("flag.zip")
    flag_zip = zipfile.ZipFile(io.BytesIO(flag_zip_data))
    password = None

    # 先试 data3~6 的所有排列
    parts = [plaintext_map[i] for i in [3, 4, 5, 6] if i in plaintext_map]
    for perm in itertools.permutations(parts):
        pwd = "".join(perm).encode()
        try:
            flag_zip.read("flag.txt", pwd=pwd)
            password = pwd
            print(f"[+] flag.zip password found: {repr(pwd.decode())}")
            break
        except Exception:
            pass

    assert password, "Could not crack flag.zip password"

    # Step 6
    flag_txt = flag_zip.read("flag.txt", pwd=password).decode("utf-8")
    print(f"[*] flag.txt visible text: {flag_txt[:20]!r}...")
    flag = decode_zwc(flag_txt)
    print()
    print(f"[!] FLAG: {flag}")


if __name__ == "__main__":
    main()
```
